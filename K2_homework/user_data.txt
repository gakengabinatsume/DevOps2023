#!/bin/bash
exec 1> >(tee -i /var/log/user-data.log) 2>&1
sudo apt update -y
sudo apt install software-properties-common
sudo add-apt-repository --yes --update ppa:ansible/ansible
sudo apt install ansible -y
sudo apt install git -y 
mkdir Ansible && cd Ansible
pwd
git clone https://github.com/mionescu75/jenkinstests.git
cd ./jenkinstests/tree/test1/playbooks/ #change to playbook folder
ansible-playbook Jenkins-playbook.yml
ansible-playbook PHP-playbook.yml
ansible-playbook Composer-playbook.yml
cd ./jenkinstests/tree/test1/ #change to compose files
composer install
